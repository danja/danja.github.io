403 void Adafruit_SSD1306::ssd1306_commandList(const uint8_t *c, uint8_t n)
404 {
405  if (wire)
406  { // I2C
407  wire->beginTransmission(i2caddr);
408  WIRE_WRITE((uint8_t)0x00); // Co = 0, D/C = 0
409  uint16_t bytesOut = 1;
410  while (n--)
411  {
412  if (bytesOut >= WIRE_MAX)
413  {
414  wire->endTransmission();
415  wire->beginTransmission(i2caddr);
416  WIRE_WRITE((uint8_t)0x00); // Co = 0, D/C = 0
417  bytesOut = 1;
418  }
419  WIRE_WRITE(pgm_read_byte(c++));
420  bytesOut++;
421  }
422  wire->endTransmission();
423  }
424  else
425  { // SPI -- transaction started in calling function
426  SSD1306_MODE_COMMAND
427  while (n--)
428  SPIwrite(pgm_read_byte(c++));
429  }

486 bool Adafruit_SSD1306::begin(uint8_t vcs, uint8_t addr, bool reset,
487  bool periphBegin)
488 {
489 
490  if ((!buffer) && !(buffer = (uint8_t *)malloc(WIDTH * ((HEIGHT + 7) / 8))))
491  return false;
492 
493  clearDisplay();
494 
495 #ifndef SSD1306_NO_SPLASH
496  if (HEIGHT > 32)
497  {
498  drawBitmap((WIDTH - splash1_width) / 2, (HEIGHT - splash1_height) / 2,
499  splash1_data, splash1_width, splash1_height, 1);
500  }
501  else
502  {
503  drawBitmap((WIDTH - splash2_width) / 2, (HEIGHT - splash2_height) / 2,
504  splash2_data, splash2_width, splash2_height, 1);
505  }
506 #endif
507 
508  vccstate = vcs;
509 
510  // Setup pin directions
511  if (wire)
512  { // Using I2C
513  // If I2C address is unspecified, use default
514  // (0x3C for 32-pixel-tall displays, 0x3D for all others).
515  i2caddr = addr ? addr : ((HEIGHT == 32) ? 0x3C : 0x3D);
516  // TwoWire begin() function might be already performed by the calling
517  // function if it has unusual circumstances (e.g. TWI variants that
518  // can accept different SDA/SCL pins, or if two SSD1306 instances
519  // with different addresses -- only a single begin() is needed).
520  if (periphBegin)
521  wire->begin();
522  }
523  else
524  { // Using one of the SPI modes, either soft or hardware
525  pinMode(dcPin, OUTPUT); // Set data/command pin as output
526  pinMode(csPin, OUTPUT); // Same for chip select
527 #ifdef HAVE_PORTREG
528  dcPort = (PortReg *)portOutputRegister(digitalPinToPort(dcPin));
529  dcPinMask = digitalPinToBitMask(dcPin);
530  csPort = (PortReg *)portOutputRegister(digitalPinToPort(csPin));
531  csPinMask = digitalPinToBitMask(csPin);
532 #endif
533  SSD1306_DESELECT
534  if (spi)
535  { // Hardware SPI
536  // SPI peripheral begin same as wire check above.
537  if (periphBegin)
538  spi->begin();
539  }
540  else
541  { // Soft SPI
542  pinMode(mosiPin, OUTPUT); // MOSI and SCLK outputs
543  pinMode(clkPin, OUTPUT);
544 #ifdef HAVE_PORTREG
545  mosiPort = (PortReg *)portOutputRegister(digitalPinToPort(mosiPin));
546  mosiPinMask = digitalPinToBitMask(mosiPin);
547  clkPort = (PortReg *)portOutputRegister(digitalPinToPort(clkPin));
548  clkPinMask = digitalPinToBitMask(clkPin);
549  *clkPort &= ~clkPinMask; // Clock low
550 #else
551  digitalWrite(clkPin, LOW); // Clock low
552 #endif
553  }
554  }
555 
556  // Reset SSD1306 if requested and reset pin specified in constructor
557  if (reset && (rstPin >= 0))
558  {
559  pinMode(rstPin, OUTPUT);
560  digitalWrite(rstPin, HIGH);
561  delay(1); // VDD goes high at start, pause for 1 ms
562  digitalWrite(rstPin, LOW); // Bring reset low
563  delay(10); // Wait 10 ms
564  digitalWrite(rstPin, HIGH); // Bring out of reset
565  }
566 
567  TRANSACTION_START
568 
569  // Init sequence
570  static const uint8_t PROGMEM init1[] = {SSD1306_DISPLAYOFF, // 0xAE
571  SSD1306_SETDISPLAYCLOCKDIV, // 0xD5
572  0x80, // the suggested ratio 0x80
573  SSD1306_SETMULTIPLEX}; // 0xA8
574  ssd1306_commandList(init1, sizeof(init1));
575  ssd1306_command1(HEIGHT - 1);
576 
577  static const uint8_t PROGMEM init2[] = {SSD1306_SETDISPLAYOFFSET, // 0xD3
578  0x0, // no offset
579  SSD1306_SETSTARTLINE | 0x0, // line #0
580  SSD1306_CHARGEPUMP}; // 0x8D
581  ssd1306_commandList(init2, sizeof(init2));
582 
583  ssd1306_command1((vccstate == SSD1306_EXTERNALVCC) ? 0x10 : 0x14);
584 
585  static const uint8_t PROGMEM init3[] = {SSD1306_MEMORYMODE, // 0x20
586  0x00, // 0x0 act like ks0108
587  SSD1306_SEGREMAP | 0x1,
588  SSD1306_COMSCANDEC};
589  ssd1306_commandList(init3, sizeof(init3));
590 
591  uint8_t comPins = 0x02;
592  contrast = 0x8F;
593 
594  if ((WIDTH == 128) && (HEIGHT == 32))
595  {
596  comPins = 0x02;
597  contrast = 0x8F;
598  }
599  else if ((WIDTH == 128) && (HEIGHT == 64))
600  {
601  comPins = 0x12;
602  contrast = (vccstate == SSD1306_EXTERNALVCC) ? 0x9F : 0xCF;
603  }
604  else if ((WIDTH == 96) && (HEIGHT == 16))
605  {
606  comPins = 0x2; // ada x12
607  contrast = (vccstate == SSD1306_EXTERNALVCC) ? 0x10 : 0xAF;
608  }
609  else
610  {
611  // Other screen varieties -- TBD
612  }
613 
614  ssd1306_command1(SSD1306_SETCOMPINS);
615  ssd1306_command1(comPins);
616  ssd1306_command1(SSD1306_SETCONTRAST);
617  ssd1306_command1(contrast);
618 
619  ssd1306_command1(SSD1306_SETPRECHARGE); // 0xd9
620  ssd1306_command1((vccstate == SSD1306_EXTERNALVCC) ? 0x22 : 0xF1);
621  static const uint8_t PROGMEM init5[] = {
622  SSD1306_SETVCOMDETECT, // 0xDB
623  0x40,
624  SSD1306_DISPLAYALLON_RESUME, // 0xA4
625  SSD1306_NORMALDISPLAY, // 0xA6
626  SSD1306_DEACTIVATE_SCROLL,
627  SSD1306_DISPLAYON}; // Main screen turn on
628  ssd1306_commandList(init5, sizeof(init5));
629 
630  TRANSACTION_END
631 
632  return true; // Success
633 }

653 void Adafruit_SSD1306::drawPixel(int16_t x, int16_t y, uint16_t color)
654 {
655  if ((x >= 0) && (x < width()) && (y >= 0) && (y < height()))
656  {
657  // Pixel is in-bounds. Rotate coordinates if needed.
658  switch (getRotation())
659  {
660  case 1:
661  ssd1306_swap(x, y);
662  x = WIDTH - x - 1;
663  break;
664  case 2:
665  x = WIDTH - x - 1;
666  y = HEIGHT - y - 1;
667  break;
668  case 3:
669  ssd1306_swap(x, y);
670  y = HEIGHT - y - 1;
671  break;
672  }
673  switch (color)
674  {
675  case SSD1306_WHITE:
676  buffer[x + (y / 8) * WIDTH] |= (1 << (y & 7));
677  break;
678  case SSD1306_BLACK:
679  buffer[x + (y / 8) * WIDTH] &= ~(1 << (y & 7));
680  break;
681  case SSD1306_INVERSE:
682  buffer[x + (y / 8) * WIDTH] ^= (1 << (y & 7));
683  break;
684  }
685  }
686 }

877 void Adafruit_SSD1306::drawFastVLineInternal(int16_t x, int16_t __y,
878  int16_t __h, uint16_t color)
879 {
880 
881  if ((x >= 0) && (x < WIDTH))
882  { // X coord in bounds?
883  if (__y < 0)
884  { // Clip top
885  __h += __y;
886  __y = 0;
887  }
888  if ((__y + __h) > HEIGHT)
889  { // Clip bottom
890  __h = (HEIGHT - __y);
891  }
892  if (__h > 0)
893  { // Proceed only if height is now positive
894  // this display doesn't need ints for coordinates,
895  // use local byte registers for faster juggling
896  uint8_t y = __y, h = __h;
897  uint8_t *pBuf = &buffer[(y / 8) * WIDTH + x];
898 
899  // do the first partial byte, if necessary - this requires some masking
900  uint8_t mod = (y & 7);
901  if (mod)
902  {
903  // mask off the high n bits we want to set
904  mod = 8 - mod;
905  // note - lookup table results in a nearly 10% performance
906  // improvement in fill* functions
907  // uint8_t mask = ~(0xFF >> mod);
908  static const uint8_t PROGMEM premask[8] = {0x00, 0x80, 0xC0, 0xE0,
909  0xF0, 0xF8, 0xFC, 0xFE};
910  uint8_t mask = pgm_read_byte(&premask[mod]);
911  // adjust the mask if we're not going to reach the end of this byte
912  if (h < mod)
913  mask &= (0XFF >> (mod - h));
914 
915  switch (color)
916  {
917  case SSD1306_WHITE:
918  *pBuf |= mask;
919  break;
920  case SSD1306_BLACK:
921  *pBuf &= ~mask;
922  break;
923  case SSD1306_INVERSE:
924  *pBuf ^= mask;
925  break;
926  }
927  pBuf += WIDTH;
928  }
929 
930  if (h >= mod)
931  { // More to go?
932  h -= mod;
933  // Write solid bytes while we can - effectively 8 rows at a time
934  if (h >= 8)
935  {
936  if (color == SSD1306_INVERSE)
937  {
938  // separate copy of the code so we don't impact performance of
939  // black/white write version with an extra comparison per loop
940  do
941  {
942  *pBuf ^= 0xFF; // Invert byte
943  pBuf += WIDTH; // Advance pointer 8 rows
944  h -= 8; // Subtract 8 rows from height
945  } while (h >= 8);
946  }
947  else
948  {
949  // store a local value to work with
950  uint8_t val = (color != SSD1306_BLACK) ? 255 : 0;
951  do
952  {
953  *pBuf = val; // Set byte
954  pBuf += WIDTH; // Advance pointer 8 rows
955  h -= 8; // Subtract 8 rows from height
956  } while (h >= 8);
957  }
958  }
959 
960  if (h)
961  { // Do the final partial byte, if necessary
962  mod = h & 7;
963  // this time we want to mask the low bits of the byte,
964  // vs the high bits we did above
965  // uint8_t mask = (1 << mod) - 1;
966  // note - lookup table results in a nearly 10% performance
967  // improvement in fill* functions
968  static const uint8_t PROGMEM postmask[8] = {0x00, 0x01, 0x03, 0x07,
969  0x0F, 0x1F, 0x3F, 0x7F};

972  {
973  case SSD1306_WHITE:
974  *pBuf |= mask;
975  break;
976  case SSD1306_BLACK:
977  *pBuf &= ~mask;
978  break;
979  case SSD1306_INVERSE:
980  *pBuf ^= mask;
981  break;
982  }
983  }
984  }
985  } // endif positive height
986  } // endif x in bounds

1001 {
1002  if ((x >= 0) && (x < width()) && (y >= 0) && (y < height()))
1003  {
1004  // Pixel is in-bounds. Rotate coordinates if needed.
1005  switch (getRotation())
1006  {
1007  case 1:
1008  ssd1306_swap(x, y);
1009  x = WIDTH - x - 1;
1010  break;
1011  case 2:
1012  x = WIDTH - x - 1;
1013  y = HEIGHT - y - 1;
1014  break;
1015  case 3:
1016  ssd1306_swap(x, y);
1017  y = HEIGHT - y - 1;
1018  break;
1019  }
1020  return (buffer[x + (y / 8) * WIDTH] & (1 << (y & 7)));
1021  }
1022  return false; // Pixel out of bounds

1041 void Adafruit_SSD1306::display(void)
1042 {
1043  TRANSACTION_START
1044  static const uint8_t PROGMEM dlist1[] = {
1045  SSD1306_PAGEADDR,
1046  0, // Page start address
1047  0xFF, // Page end (not really, but works here)
1048  SSD1306_COLUMNADDR, 0}; // Column start address
1049  ssd1306_commandList(dlist1, sizeof(dlist1));
1050  ssd1306_command1(WIDTH - 1); // Column end address
1051 
1052 #if defined(ESP8266)
1053  // ESP8266 needs a periodic yield() call to avoid watchdog reset.
1054  // With the limited size of SSD1306 displays, and the fast bitrate
1055  // being used (1 MHz or more), I think one yield() immediately before
1056  // a screen write and one immediately after should cover it. But if
1057  // not, if this becomes a problem, yields() might be added in the
1058  // 32-byte transfer condition below.
1059  yield();
1060 #endif
1061  uint16_t count = WIDTH * ((HEIGHT + 7) / 8);
1062  uint8_t *ptr = buffer;
1063  if (wire)
1064  { // I2C
1065  wire->beginTransmission(i2caddr);
1066  WIRE_WRITE((uint8_t)0x40);
1067  uint16_t bytesOut = 1;
1068  while (count--)
1069  {
1070  if (bytesOut >= WIRE_MAX)
1071  {
1072  wire->endTransmission();
1073  wire->beginTransmission(i2caddr);
1074  WIRE_WRITE((uint8_t)0x40);
1075  bytesOut = 1;
1076  }
1077  WIRE_WRITE(*ptr++);
1078  bytesOut++;
1079  }
1080  wire->endTransmission();
1081  }
1082  else
1083  { // SPI
1084  SSD1306_MODE_DATA
1085  while (count--)
1086  SPIwrite(*ptr++);
1087  }
1088  TRANSACTION_END
1089 #if defined(ESP8266)
1090  yield();
1091 #endif
1092 }
