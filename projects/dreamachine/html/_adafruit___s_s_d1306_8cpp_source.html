416 void Adafruit_SSD1306::ssd1306_commandList(const uint8_t *c, uint8_t n) {
417  if (wire) { // I2C
418  wire->beginTransmission(i2caddr);
419  WIRE_WRITE((uint8_t)0x00); // Co = 0, D/C = 0

642  if ((x >= 0) && (x < width()) && (y >= 0) && (y < height())) {
643  // Pixel is in-bounds. Rotate coordinates if needed.
644  switch (getRotation()) {
645  case 1:
646  ssd1306_swap(x, y);
647  x = WIDTH - x - 1;
648  break;
649  case 2:
650  x = WIDTH - x - 1;
651  y = HEIGHT - y - 1;
652  break;
653  case 3:
654  ssd1306_swap(x, y);
655  y = HEIGHT - y - 1;
656  break;
657  }
658  switch (color) {
659  case SSD1306_WHITE:
660  buffer[x + (y / 8) * WIDTH] |= (1 << (y & 7));
661  break;
662  case SSD1306_BLACK:
663  buffer[x + (y / 8) * WIDTH] &= ~(1 << (y & 7));
664  break;
665  case SSD1306_INVERSE:
666  buffer[x + (y / 8) * WIDTH] ^= (1 << (y & 7));

847 void Adafruit_SSD1306::drawFastVLineInternal(int16_t x, int16_t __y,
848  int16_t __h, uint16_t color) {
849 
850  if ((x >= 0) && (x < WIDTH)) { // X coord in bounds?
851  if (__y < 0) { // Clip top
852  __h += __y;
853  __y = 0;
854  }
855  if ((__y + __h) > HEIGHT) { // Clip bottom
856  __h = (HEIGHT - __y);
857  }
858  if (__h > 0) { // Proceed only if height is now positive
859  // this display doesn't need ints for coordinates,
860  // use local byte registers for faster juggling
861  uint8_t y = __y, h = __h;
862  uint8_t *pBuf = &buffer[(y / 8) * WIDTH + x];
863 
864  // do the first partial byte, if necessary - this requires some masking
865  uint8_t mod = (y & 7);
866  if (mod) {
867  // mask off the high n bits we want to set
868  mod = 8 - mod;
869  // note - lookup table results in a nearly 10% performance
870  // improvement in fill* functions
871  // uint8_t mask = ~(0xFF >> mod);
872  static const uint8_t PROGMEM premask[8] = {0x00, 0x80, 0xC0, 0xE0,
873  0xF0, 0xF8, 0xFC, 0xFE};
874  uint8_t mask = pgm_read_byte(&premask[mod]);
875  // adjust the mask if we're not going to reach the end of this byte
876  if (h < mod)
877  mask &= (0XFF >> (mod - h));
878 
879  switch (color) {
880  case SSD1306_WHITE:
881  *pBuf |= mask;
882  break;
883  case SSD1306_BLACK:
884  *pBuf &= ~mask;
885  break;
886  case SSD1306_INVERSE:
887  *pBuf ^= mask;
888  break;
889  }
890  pBuf += WIDTH;
891  }
892 
893  if (h >= mod) { // More to go?
894  h -= mod;
895  // Write solid bytes while we can - effectively 8 rows at a time
896  if (h >= 8) {
897  if (color == SSD1306_INVERSE) {
898  // separate copy of the code so we don't impact performance of
899  // black/white write version with an extra comparison per loop
900  do {
901  *pBuf ^= 0xFF; // Invert byte
902  pBuf += WIDTH; // Advance pointer 8 rows
903  h -= 8; // Subtract 8 rows from height
904  } while (h >= 8);
905  } else {
906  // store a local value to work with
907  uint8_t val = (color != SSD1306_BLACK) ? 255 : 0;
908  do {
909  *pBuf = val; // Set byte
910  pBuf += WIDTH; // Advance pointer 8 rows
911  h -= 8; // Subtract 8 rows from height
912  } while (h >= 8);
913  }
914  }
915 
916  if (h) { // Do the final partial byte, if necessary
917  mod = h & 7;
918  // this time we want to mask the low bits of the byte,
919  // vs the high bits we did above
920  // uint8_t mask = (1 << mod) - 1;
921  // note - lookup table results in a nearly 10% performance
922  // improvement in fill* functions
923  static const uint8_t PROGMEM postmask[8] = {0x00, 0x01, 0x03, 0x07,
924  0x0F, 0x1F, 0x3F, 0x7F};
925  uint8_t mask = pgm_read_byte(&postmask[mod]);
926  switch (color) {
927  case SSD1306_WHITE:
928  *pBuf |= mask;
929  break;
930  case SSD1306_BLACK:
931  *pBuf &= ~mask;
932  break;
933  case SSD1306_INVERSE:
934  *pBuf ^= mask;
935  break;
936  }
937  }
938  }
939  } // endif positive height
940  } // endif x in bounds
941 }

992 void Adafruit_SSD1306::display(void) {
993  TRANSACTION_START
994  static const uint8_t PROGMEM dlist1[] = {
995  SSD1306_PAGEADDR,
996  0, // Page start address
997  0xFF, // Page end (not really, but works here)
998  SSD1306_COLUMNADDR, 0}; // Column start address
999  ssd1306_commandList(dlist1, sizeof(dlist1));
1000  ssd1306_command1(WIDTH - 1); // Column end address
1001 
1002 #if defined(ESP8266)
1003  // ESP8266 needs a periodic yield() call to avoid watchdog reset.
1004  // With the limited size of SSD1306 displays, and the fast bitrate
1005  // being used (1 MHz or more), I think one yield() immediately before
1006  // a screen write and one immediately after should cover it. But if
1007  // not, if this becomes a problem, yields() might be added in the
1008  // 32-byte transfer condition below.
1009  yield();
1010 #endif
1011  uint16_t count = WIDTH * ((HEIGHT + 7) / 8);
1012  uint8_t *ptr = buffer;
1013  if (wire) { // I2C
1014  wire->beginTransmission(i2caddr);
1015  WIRE_WRITE((uint8_t)0x40);
1016  uint16_t bytesOut = 1;
1017  while (count--) {
1018  if (bytesOut >= WIRE_MAX) {
1019  wire->endTransmission();
1020  wire->beginTransmission(i2caddr);
1021  WIRE_WRITE((uint8_t)0x40);
1022  bytesOut = 1;
1023  }
1024  WIRE_WRITE(*ptr++);
1025  bytesOut++;
1026  }
1027  wire->endTransmission();
1028  } else { // SPI
1029  SSD1306_MODE_DATA
1030  while (count--)
1031  SPIwrite(*ptr++);
1032  }
1033  TRANSACTION_END
1034 #if defined(ESP8266)
1035  yield();
1036 #endif
1037 }
